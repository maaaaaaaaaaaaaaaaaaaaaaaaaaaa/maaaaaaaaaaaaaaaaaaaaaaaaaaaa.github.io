import turtle

turtle.register_shape("0926/bug.gif")

소크라테스=turtle.Turtle()
소크라테스.shape("0926/bug.gif")
한변=100

for _ in range(3):
    소크라테스.forward(100)
    소크라테스.right(120)

소크라테스.forward(한변/3)
소크라테스.left(60)
소크라테스.forward(한변/3)

for _ in range(3):
    소크라테스.right(120)
    소크라테스.forward(100)
     

turtle.done()
